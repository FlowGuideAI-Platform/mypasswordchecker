    return new Date() < this.PROMO_END_DATE;
  },

    return this.isPromoActive() ? this.PROMO_PRICE : this.ORIGINAL_PRICE;
  },
  
  // Days remaining in promotion
  getDaysRemaining() {
    const now = new Date();
    const diff = this.PROMO_END_DATE - now;
    return Math.max(0, Math.ceil(diff / (1000 * 60 * 60 * 24)));
  }
};

### B) Implement Grandfathered Pricing Logic

**File:** `src/js/api-auth.js` (NEW/MODIFICATION)
````javascript
/**
 * API Subscription Management
 * Handles promotional pricing and grandfathering
 */

import PROMO_CONFIG from './config.js';

class APISubscription {
  constructor(userId, tier, signupDate) {
    this.userId = userId;
    this.tier = tier;
    this.signupDate = new Date(signupDate);
  }
  
  /**
   * Determine pricing for this subscription
   * Users who signed up during promo get grandfathered rates
   */
  getCurrentPrice() {
    const prices = PROMO_CONFIG.getCurrentPrices();
    
    // Check if user signed up during promotional period
    if (this.signupDate < PROMO_CONFIG.PROMO_END_DATE) {
      // Grandfathered at promotional rate
      return PROMO_CONFIG.PROMO_PRICE[this.tier];
    }
    
    // New signups after promo pay full price
    return prices[this.tier];
  }
  
  /**
   * Check if user is grandfathered
   */
  isGrandfathered() {
    return this.signupDate < PROMO_CONFIG.PROMO_END_DATE;
  }
  
  /**
   * Get savings amount if grandfathered
   */
  getMonthlySavings() {
    if (!this.isGrandfathered()) return 0;
    
    const fullPrice = PROMO_CONFIG.ORIGINAL_PRICE[this.tier];
    const myPrice = PROMO_CONFIG.PROMO_PRICE[this.tier];
    
    return fullPrice - myPrice;
  }
}

export default APISubscription;
````

## How It Works

### Client-Side Architecture

All password analysis runs entirely in your browser:
```javascript
// Your password never leaves your device
function analyzePassword(password) {
  const strength = zxcvbn(password);           // Pattern detection
  const entropy = calculateEntropy(password);   // Randomness measure
  const quantum = estimateQuantum(entropy);     // Grover's algorithm
  const breached = await checkHIBP(password);   // k-anonymity check
  
  return { strength, entropy, quantum, breached };
}
```

**Privacy guarantee:** Works offline. Nothing is sent to servers from browser.

/**
 * Automatic price reversion on April 1, 2026
 * 
 * This script checks daily and updates UI/pricing when promo ends.
 * No manual intervention required.
 */

import PROMO_CONFIG from './config.js';

class PriceManager {
  constructor() {
    this.checkInterval = 1000 * 60 * 60; // Check every hour
    this.init();
  }
  
  init() {
    // Check immediately on page load
    this.checkAndUpdate();
    
    // Then check periodically
    setInterval(() => this.checkAndUpdate(), this.checkInterval);
  }
  
  checkAndUpdate() {
    const isPromoActive = PROMO_CONFIG.isPromoActive();
    
    if (!isPromoActive && this.wasPromoActive) {
      // Promo just ended - trigger UI updates
      this.revertToFullPricing();
    }
    
    this.wasPromoActive = isPromoActive;
  }
  
  revertToFullPricing() {
    console.log('Promotional period ended - reverting to full pricing');
    
    // Update web interface
    this.updateWebPricing();
    
    // Update API pricing display
    this.updateAPIPricing();
    
    // Show notification to users
    this.showPriceChangeNotification();
  }
  
  updateWebPricing() {
    // Hide promotional banner
    const promoBanner = document.getElementById('promo-banner');
    if (promoBanner) {
      promoBanner.style.display = 'none';
    }
    
    // Update feature access to require payment
    // (isPremiumAccessible() will now return false for non-paying users)
    
    // Show "Premium features now $1" messaging
    const features = document.querySelectorAll('.premium-feature');
    features.forEach(feature => {
      feature.classList.add('requires-payment');
      const badge = document.createElement('span');
      badge.className = 'premium-badge';
      badge.textContent = '$1.00';
      feature.appendChild(badge);
    });
  }
  
  updateAPIPricing() {
    // Update pricing cards to show full prices
    document.querySelectorAll('.promo-price').forEach(el => {
      el.style.display = 'none';
    });
    
    document.querySelectorAll('.original-price').forEach(el => {
      el.style.textDecoration = 'none';
      el.classList.remove('original-price');
      el.classList.add('current-price');
    });
    
    // Remove promotional savings badges
    document.querySelectorAll('.promo-savings').forEach(el => {
      el.remove();
    });
  }
  
  showPriceChangeNotification() {
    // Optional: Show a banner informing users
    const notice = document.createElement('div');
    notice.className = 'price-change-notice';
    notice.innerHTML = `
      <p>
        📢 The promotional period has ended. 
        Premium features are now $1.00 (one-time payment).
      </p>
      <p>
        <strong>Grandfathered users:</strong> Your promotional rate is preserved!
      </p>
      <button onclick="this.parentElement.remove()">Got it</button>
    `;
    document.body.insertBefore(notice, document.body.firstChild);
  }
}

if (typeof window !== 'undefined') {
  new PriceManager();

/**

class PromoAnalytics {
  constructor() {

  }

  trackPromoFeatureUsage(feature) {

    });
  }
  
  /**
   * Track when promo banner is shown vs hidden
   */
  trackPromoBannerImpression() {
    if (PROMO_CONFIG.isPromoActive()) {
      this.logEvent('promo_banner_shown', {
        days_remaining: PROMO_CONFIG.getDaysRemaining()
      });
    }
  }
  
  /**
   * Log event (send to your analytics endpoint)
   */
  logEvent(event_name, properties) {
    // Send to Cloudflare Analytics or your preferred service
    // Ensure no PII is included
    
    if (typeof fetch !== 'undefined') {
      fetch('/api/analytics', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          event: event_name,
          properties: properties,
          // No user identifiers - just aggregate data
        })
      }).catch(() => {
        // Silently fail - analytics shouldn't break functionality
      });
    }
  }
}
